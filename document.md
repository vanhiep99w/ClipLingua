/**

async function processWithGroq(text, apiKey) {
  const language = detectLanguage(text);
  const prompt =
    language === "en"
      ? PROMPT_ENGLISH.replace("{text}", text)
      : PROMPT_VIETNAMESE.replace("{text}", text);

  try {
    const response = await fetch(GROQ_API_ENDPOINT, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${apiKey}`,
      },
      body: JSON.stringify({
        model: MODEL,
        messages: [{ role: "user", content: prompt }],
        temperature: API_CONFIG.temperature,
        max_tokens: API_CONFIG.max_tokens,
      }),
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`);
    }

    const data = await response.json();
    const result = data.choices[0].message.content;

    // Parse result
    if (language === "en") {
      return parseEnglishResult(result);
    } else {
      return { translated: result.trim() };
    }
  } catch (error) {
    console.error("Groq API Error:", error);
    throw error;
  }
}

/**
 * Parse káº¿t quáº£ tá»« English prompt
 */
function parseEnglishResult(result) {
  const lines = result.split("\n");
  let corrected = "";
  let translated = "";

  for (const line of lines) {
    if (line.startsWith("CORRECTED:")) {
      corrected = line.replace("CORRECTED:", "").trim();
    } else if (line.startsWith("VIETNAMESE:")) {
      translated = line.replace("VIETNAMESE:", "").trim();
    }
  }

  return { corrected, translated };
}
